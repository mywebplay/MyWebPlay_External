<!-- Call secure API -->
<script>
function CallSecureApi(s) {
  fetch("/Cover/SecureWithCallApi", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(s)
  })
  .then(res => res.text()) // lấy raw text trước
  .then(text => {
    try {
      console.log(JSON.parse(text)); // thử parse JSON
    } catch {
      console.log(text); // nếu không phải JSON thì in text
    }
  })
  .catch(console.error);
}
</script>
