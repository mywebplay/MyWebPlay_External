<div id="dtModal" style="display:none; position:fixed; inset:0; background:rgba(0,0,0,0.5);">
  <div style="background:#fff; width:320px; margin:10% auto; padding:20px; border-radius:8px;">
    
    <input id="dtPicker" />

    <div style="margin-top:10px;">
      <label>
        <input type="checkbox" id="chkUseTime"> Chọn giờ
      </label>
    </div>

    <div style="margin-top:15px; text-align:right;">
      <button id="btnCancel" style="background:red; color:#fff;">Huỷ</button>
      <button id="btnOk" style="background:green; color:#fff;">Xác nhận</button>
    </div>

  </div>

<script>
const placeholders = [
    "Nhập từ khoá tìm kiếm theo địa chỉ IP",
    "Nhập từ khoá tìm kiếm theo thông tin user",
    "Nhập từ khoá tìm kiếm theo ngày truy cập lần đầu (bắt đầu, định dạng: yyyy-MM-dd HH:mm:ss)",
    "Nhập từ khoá tìm kiếm theo ngày truy cập lần đầu (kết thúc, định dạng: yyyy-MM-dd HH:mm:ss)",
    "Nhập từ khoá tìm kiếm theo mã hoá nhận danh",
    "Nhập từ khoá tìm kiếm theo tiền kiếp IPs",
];

for (let i = 0; i < placeholders.length; i++) {
    document.getElementById("txtParam" + i).placeholder = placeholders[i];
    document.getElementById("txtParam" + i).title = placeholders[i];
}

document.getElementById("txtType2").selectedIndex = 5;
document.getElementById("txtType3").selectedIndex = 2;

document.getElementById("txtParam2").readOnly = true;
document.getElementById("txtParam3").readOnly = true;

(function () {
    function removeHash() {
        if (window.location.hash) {
            history.replaceState(
                null,
                document.title,
                window.location.pathname + window.location.search
            );
        }
    }

    // chạy ngay khi load
    removeHash();

    // nếu fragment thay đổi (user click anchor, script gán...)
    window.addEventListener("hashchange", removeHash);

    // backup thêm (nếu có script khác set hash liên tục)
    setInterval(removeHash, 500);
})();

// ===== GLOBAL =====
let currentInput = null;
let picker = null;

// ===== BIND TEXTBOX =====
function bindDateTimePicker(id) {
    const el = document.getElementById(id);
    if (!el) return;

    el.addEventListener('click', function () {
        openDateTimeModal(el);
    });
}

// ===== OPEN MODAL =====
function openDateTimeModal(input) {
    currentInput = input;

    const modal = document.getElementById('dtModal');
    const chk = document.getElementById('chkUseTime');

    modal.style.display = 'block';

    if (picker) picker.destroy();

    // detect value cũ có giờ hay không
    const hasTime = input.value && input.value.includes(':');

    chk.checked = hasTime;

    picker = flatpickr("#dtPicker", {
        enableTime: hasTime,
        dateFormat: "Y-m-d H:i:S",
        time_24hr: true,
        defaultDate: input.value || null
    });
}

// ===== TOGGLE TIME =====
document.addEventListener('change', function (e) {
    if (e.target && e.target.id === 'chkUseTime') {

        if (!picker) return;

        const selected = picker.selectedDates[0];

        picker.destroy();

        picker = flatpickr("#dtPicker", {
            enableTime: e.target.checked,
            dateFormat: "Y-m-d H:i:S",
            time_24hr: true,
            defaultDate: selected || null
        });
    }
});

// ===== CLICK EVENTS (OK / CANCEL / BACKDROP) =====
document.addEventListener('click', function (e) {

    // OK
    if (e.target.closest('#btnOk')) {

        if (!picker || !currentInput) return;

        const selected = picker.selectedDates[0];
        if (!selected) {
            closeModal();
            return;
        }

        const useTime = document.getElementById('chkUseTime').checked;

        let value = useTime
            ? formatDateTime(selected)
            : formatDate(selected);

        currentInput.value = value;

        closeModal();
    }

    // CANCEL
    if (e.target.closest('#btnCancel')) {
        closeModal();
    }

    // CLICK OUTSIDE
    if (e.target.id === 'dtModal') {
        closeModal();
    }
});

// ===== ESC CLOSE =====
document.addEventListener('keydown', function (e) {
    if (e.key === 'Escape') {
        closeModal();
    }
});

// ===== CLOSE MODAL =====
function closeModal() {
    const modal = document.getElementById('dtModal');
    if (modal) modal.style.display = 'none';
}

// ===== FORMAT DATE =====
function formatDate(d) {
    const yyyy = d.getFullYear();
    const MM = String(d.getMonth() + 1).padStart(2, '0');
    const dd = String(d.getDate()).padStart(2, '0');

    return `${yyyy}-${MM}-${dd}`;
}

// ===== FORMAT DATETIME =====
function formatDateTime(d) {
    const yyyy = d.getFullYear();
    const MM = String(d.getMonth() + 1).padStart(2, '0');
    const dd = String(d.getDate()).padStart(2, '0');

    const HH = String(d.getHours()).padStart(2, '0');
    const mm = String(d.getMinutes()).padStart(2, '0');
    const ss = String(d.getSeconds()).padStart(2, '0');

    return `${yyyy}-${MM}-${dd} ${HH}:${mm}:${ss}`;
}

bindDateTimePicker('txtParam2');
bindDateTimePicker('txtParam3');
</script>
