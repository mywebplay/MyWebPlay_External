function safeInsert(quill, text) {
    const clean = text.replace(/\r\n/g, "\n");

    // 🔥 detach hoàn toàn khỏi Quill event loop
    Promise.resolve().then(() => {
        const len = quill.getLength();

        quill.setSelection(len, 0);

        quill.insertText(len, clean, 'user');
    });
}

function lockDeleteInQuill(quill) {

    quill.root.addEventListener("keydown", function (e) {

        if (
            e.key === "Backspace" ||
            e.key === "Delete"
        ) {
            e.preventDefault();
            e.stopPropagation();
        }

    }, true);
}

function clearQuill() {
    quill.setContents([]);
    quill.setSelection(0, 0);
}

function lockBackspace(quill) {

    quill.keyboard.bindings[8] = []; // 8 = Backspace
    quill.keyboard.bindings[46] = []; // 46 = Delete (optional)
}

lockDeleteInQuill(quill);
lockBackspace(quill);
