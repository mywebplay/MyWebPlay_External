<script>
const select = document.getElementById("fixedSelect");

let hasSelected = false;

// chọn -> nhảy
select.onchange = function () {
    if (!this.value) return;

    hasSelected = true;

    window.location.href = "/Admin/SettingXYZ_DarkAdmin#" + this.value;
};

// nếu KHÔNG chọn gì mà chỉ hover rồi rời khỏi dropdown -> reset
select.onblur = function () {
    if (hasSelected) return; // đã từng chọn thì không reset nữa

    this.selectedIndex = 0;
};
</script>
