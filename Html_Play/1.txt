<option value="come-39">Sửa email/megaIO</option>
<option value="come-52">Upload file max size</option>
<option value="custom180">Bảng thông tin</option>
<option value="changesetting">Setting secure admin</option>
</select>

<script>
const select = document.getElementById("fixedSelect");

let waitingReset = false;

// chọn -> chỉ điều hướng, KHÔNG reset
select.onchange = function () {
    const val = this.value;
    if (!val) return;

    waitingReset = true;

    window.location.href = "/Admin/SettingXYZ_DarkAdmin#" + val;
};

// reset khi user bắt đầu tương tác lại
function resetSelectOnce() {
    if (!waitingReset) return;

    select.selectedIndex = 0;
    waitingReset = false;

    window.removeEventListener("scroll", onScroll);
    document.removeEventListener("click", onClickOutside);
}

function onScroll() {
    resetSelectOnce();
}

function onClickOutside(e) {
    if (e.target === select) return;
    resetSelectOnce();
}

// chỉ kích hoạt listener sau khi đã chọn
window.addEventListener("scroll", onScroll, { passive: true });
document.addEventListener("click", onClickOutside);
</script>
