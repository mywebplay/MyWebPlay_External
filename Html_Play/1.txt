<script>
const select = document.getElementById("fixedSelect");

select.onchange = function () {
    const val = this.value;
    if (!val) return;

    // reset ngay UI về mặc định (tránh giữ selected)
    this.selectedIndex = 0;

    // điều hướng sau 1 tick để đảm bảo reset UI trước
    setTimeout(() => {
        window.location.href = "/Admin/SettingXYZ_DarkAdmin#" + val;
    }, 0);
};
</script>
