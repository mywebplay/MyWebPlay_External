<script>
const select = document.getElementById("fixedSelect");

let lastVal = null;

// chọn -> điều hướng, KHÔNG reset ngay
select.onchange = function () {
    if (!this.value) return;

    lastVal = this.value;

    window.location.href = "/Admin/SettingXYZ_DarkAdmin#" + lastVal;
};

// khi hash load xong -> reset dropdown
window.addEventListener("load", () => {
    if (!lastVal) return;

    setTimeout(() => {
        select.selectedIndex = 0;
        lastVal = null;
    }, 300);
});
</script>
