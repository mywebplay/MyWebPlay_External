#fixedSelect {
position: fixed;
bottom: 10px;
right: 10px;
z-index: 9999;
}
</style>

<select id="fixedSelect">
<option value=""></option>
<option value="linkread6">Lên đầu trang</option>
<option value="come-39">Sửa email/megaIO</option>
<option value="come-52">Upload file max size</option>
<option value="custom180">Bảng thông tin</option>
<option value="changesetting">Setting secure admin</option>
</select>

<script>
const select = document.getElementById("fixedSelect");

let resetDone = false;

// chọn -> nhảy tới #x (GIỮ nguyên selected)
select.onchange = function () {
    const val = this.value;
    if (!val) return;

    resetDone = false;

    window.location.href = "/Admin/SettingXYZ_DarkAdmin#" + val;
};

// click ra ngoài -> reset
document.addEventListener("click", function (e) {
    if (resetDone) return;
    if (e.target === select) return;

    select.selectedIndex = 0;
    resetDone = true;
});

// scroll -> reset
window.addEventListener("scroll", function () {
    if (resetDone) return;

    select.selectedIndex = 0;
    resetDone = true;
});
</script>
