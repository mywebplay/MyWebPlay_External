function go(id) {
window.location.href = "/Admin/SettingXYZ_DarkAdmin#" + id;
}

// highlight section hiện tại theo hash
function updateActive() {
const hash = location.hash.replace("#", "");

document.querySelectorAll("#quickNav div").forEach(el => {
el.classList.remove("active");
});

if (hash) {
const active = document.getElementById("nav-" + hash);
if (active) active.classList.add("active");
}
}

window.addEventListener("hashchange", updateActive);
window.addEventListener("load", updateActive);
</script>
