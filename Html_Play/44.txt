<script>

var listFileCanRemove = ["/ClientConnect/ListIPComeHere.txt", "/Admin/ListIPComeToHereTheFirst.txt"];

(function () {

    if (document.getElementById("mwp_hp_remove_line_initialized"))
    {
        return;
    }

    var initFlag = document.createElement("div");
    initFlag.id = "mwp_hp_remove_line_initialized";
    initFlag.style.display = "none";
    document.body.appendChild(initFlag);

    function init()
    {
        var txtFile = document.getElementById("filene");
        var txtData = document.getElementById("data");

        if (!txtFile || !txtData)
        {
            return;
        }

        if (listFileCanRemove.indexOf(txtFile.value) < 0)
        {
            return;
        }

        var root = document.getElementById("mwp_hp_remove_line_root");
        var btn = document.getElementById("mwp_hp_remove_line_btn");

        if (!root || !btn)
        {
            return;
        }

        root.style.display = "";

        if (!btn.parentNode.__mwp_inserted)
        {
            txtData.insertAdjacentElement("afterend", btn.parentNode);
            btn.parentNode.__mwp_inserted = true;
        }

        bindEvent();
    }

    var binded = false;
    var lineData = [];

    function bindEvent()
    {
        if (binded)
        {
            return;
        }

        binded = true;

        document.getElementById("mwp_hp_remove_line_btn").addEventListener("click", openModal);

        document.getElementById("mwp_hp_remove_line_close").addEventListener("click", closeModal);

        document.getElementById("mwp_hp_remove_line_confirm").addEventListener("click", confirmData);

        document.getElementById("mwp_hp_remove_line_list").addEventListener("click", function (e)
        {
            if (!e.target.classList.contains("mwp_hp_remove_line_remove"))
            {
                return;
            }

            var index = parseInt(e.target.parentNode.dataset.index);

            removeItem(index);
        });
    }

    function openModal()
    {
        document.getElementById("mwp_hp_remove_line_root").style.display = "block";
        var txtData = document.getElementById("data");

        lineData = txtData.value
            .split(/\r\n|\n|\r/)
            .filter(function (x)
            {
                return x.trim() !== "";
            });

        renderList();

        document.getElementById("mwp_hp_remove_line_overlay").style.display = "block";

        document.body.style.overflow = "hidden";
    }

    function closeModal()
    {
        document.getElementById("mwp_hp_remove_line_overlay").style.display = "none";

        document.body.style.overflow = "";
    }

    function renderList()
    {
        var list = document.getElementById("mwp_hp_remove_line_list");

        list.innerHTML = "";

        for (var i = 0; i < lineData.length; i++)
        {
            var item = document.createElement("div");

            item.className = "mwp_hp_remove_line_item";

            item.dataset.index = i;

            var text = lineData[i];

            if (text.length > 300)
            {
                text = text.substring(0, 300) + "...";
            }

            item.innerHTML =
                '<div class="mwp_hp_hp_remove_line_text">' +
                escapeHtml(text) +
                '</div>' +
                (i === 0
                    ? ''
                    : '<div class="mwp_hp_remove_line_remove">×</div>');

            list.appendChild(item);
        }
    }

    function removeItem(index)
    {
        var list = document.getElementById("mwp_hp_remove_line_list");

        var item = list.querySelector('[data-index="' + index + '"]');

        if (!item)
        {
            return;
        }

        item.classList.add("mwp_hp_remove_line_item_remove");

        setTimeout(function ()
        {
            lineData.splice(index, 1);

            renderList();

        }, 180);
    }

    function confirmData()
    {
        document.getElementById("data").value = lineData.join("\r\n");

        closeModal();
    }

    function escapeHtml(text)
    {
        return text
            .replace(/&/g, "&amp;")
            .replace(/</g, "&lt;")
            .replace(/>/g, "&gt;")
            .replace(/"/g, "&quot;")
            .replace(/'/g, "&#39;");
    }

    if (document.readyState === "loading")
    {
        document.addEventListener("DOMContentLoaded", init);
    }
    else
    {
        init();
    }

})();

</script>
