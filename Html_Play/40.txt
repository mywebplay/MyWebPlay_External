<!-- Thiết kế mở rộng/thu gọn textarea  -->
<style>
textarea {
    box-sizing: border-box;
    resize: none;
    min-width: 200px;
    min-height: 100px;
}

/* wrapper là thứ chịu resize */
.textarea-wrap {
    position: relative;
    display: inline-block;

    /* QUAN TRỌNG: tránh flex/grid bó */
    flex: none;
}

/* textarea luôn fill wrapper */
.textarea-wrap > textarea {
    width: 100% !important;
    height: 100% !important;
}

/* handle chung */
.handle {
    position: absolute;
    z-index: 10;
    opacity: 0;
    transition: opacity .15s;
}

/* hover mới hiện */
.textarea-wrap:hover .handle {
    opacity: 1;
}

/* edges */
.h-n {
    top: -4px; left: 6px; right: 6px;
    height: 8px;
    cursor: n-resize;
}

.h-s {
    bottom: -4px; left: 6px; right: 6px;
    height: 8px;
    cursor: s-resize;
}

.h-e {
    top: 6px; bottom: 6px; right: -4px;
    width: 8px;
    cursor: e-resize;
}

.h-w {
    top: 6px; bottom: 6px; left: -4px;
    width: 8px;
    cursor: w-resize;
}

/* corners */
.h-nw, .h-ne, .h-sw, .h-se {
    width: 12px;
    height: 12px;
}

.h-nw { top: -5px; left: -5px; cursor: nw-resize; }
.h-ne { top: -5px; right: -5px; cursor: ne-resize; }
.h-sw { bottom: -5px; left: -5px; cursor: sw-resize; }
.h-se { bottom: -5px; right: -5px; cursor: se-resize; }
</style>
<script>
(() => {

    const MIN_W = 200;
    const MIN_H = 100;

    const DIRS = ["n","s","e","w","nw","ne","sw","se"];

    function makeResizable(textarea) {

        if (textarea.dataset.resizableDone) return;
        textarea.dataset.resizableDone = "1";

        const wrap = document.createElement("div");
        wrap.className = "textarea-wrap";

        // lấy size ban đầu chuẩn
        const rect = textarea.getBoundingClientRect();

        wrap.style.width = rect.width + "px";
        wrap.style.height = rect.height + "px";

        // FIX QUAN TRỌNG: tránh bị flex/grid override
        wrap.style.display = "inline-block";
        wrap.style.flex = "none";

        textarea.parentNode.insertBefore(wrap, textarea);
        wrap.appendChild(textarea);

        // đảm bảo fill wrapper
        textarea.style.width = "100%";
        textarea.style.height = "100%";

        DIRS.forEach(dir => {

            const h = document.createElement("div");
            h.className = `handle h-${dir}`;
            wrap.appendChild(h);

            h.addEventListener("mousedown", (e) => {

                e.preventDefault();

                const start = {
                    x: e.clientX,
                    y: e.clientY,
                    w: wrap.offsetWidth,
                    h: wrap.offsetHeight
                };

                function move(ev) {

                    let newW = start.w;
                    let newH = start.h;

                    const dx = ev.clientX - start.x;
                    const dy = ev.clientY - start.y;

                    if (dir.includes("e")) newW = start.w + dx;
                    if (dir.includes("s")) newH = start.h + dy;
                    if (dir.includes("w")) newW = start.w - dx;
                    if (dir.includes("n")) newH = start.h - dy;

                    newW = Math.max(MIN_W, newW);
                    newH = Math.max(MIN_H, newH);

                    wrap.style.width = newW + "px";
                    wrap.style.height = newH + "px";
                }

                function stop() {
                    document.removeEventListener("mousemove", move);
                    document.removeEventListener("mouseup", stop);
                }

                document.addEventListener("mousemove", move);
                document.addEventListener("mouseup", stop);
            });
        });
    }

    function init() {
        document.querySelectorAll("textarea")
            .forEach(makeResizable);
    }

    init();

    new MutationObserver(init)
        .observe(document.body, {
            childList: true,
            subtree: true
        });

})();
</script>
