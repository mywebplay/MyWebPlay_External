<!-- Thiết kế mở rộng/thu gọn textarea  -->
<style>
textarea {
    min-width: 200px;
    min-height: 100px;

    resize: none;
    overflow: auto;

    box-sizing: border-box;
}

.textarea-resize-wrapper {
    position: relative;
    display: inline-block;
}

.textarea-resizing {
    user-select: none;
}

.resize-handle {
    position: absolute;
    opacity: 0;
    z-index: 1000;
    transition: opacity .15s;
}

.textarea-resize-wrapper:hover .resize-handle,
.textarea-resizing ~ .resize-handle,
.textarea-resize-wrapper:hover textarea {
    opacity: 1;
}

.textarea-resize-wrapper:hover textarea,
.textarea-resizing {
    box-shadow: 0 0 0 3px rgba(0,0,0,.08);
}

/* cạnh */

.resize-n {
    top: -4px;
    left: 10px;
    right: 10px;
    height: 8px;
    cursor: n-resize;
}

.resize-s {
    bottom: -4px;
    left: 10px;
    right: 10px;
    height: 8px;
    cursor: s-resize;
}

.resize-w {
    top: 10px;
    bottom: 10px;
    left: -4px;
    width: 8px;
    cursor: w-resize;
}

.resize-e {
    top: 10px;
    bottom: 10px;
    right: -4px;
    width: 8px;
    cursor: e-resize;
}

/* góc */

.resize-nw {
    width: 14px;
    height: 14px;
    top: -5px;
    left: -5px;
    cursor: nw-resize;
}

.resize-ne {
    width: 14px;
    height: 14px;
    top: -5px;
    right: -5px;
    cursor: ne-resize;
}

.resize-sw {
    width: 14px;
    height: 14px;
    bottom: -5px;
    left: -5px;
    cursor: sw-resize;
}

.resize-se {
    width: 14px;
    height: 14px;
    bottom: -5px;
    right: -5px;
    cursor: se-resize;
}
</style>
<script>
(() => {

    const MIN_WIDTH = 200;
    const MIN_HEIGHT = 100;

    const HANDLES = [
        "n","s","e","w",
        "nw","ne","sw","se"
    ];

    function makeResizable(textarea) {

        if (textarea.dataset.resizeReady) return;
        textarea.dataset.resizeReady = "1";

        const wrapper = document.createElement("div");
        wrapper.className = "textarea-resize-wrapper";

        textarea.parentNode.insertBefore(wrapper, textarea);
        wrapper.appendChild(textarea);

        HANDLES.forEach(dir => {

            const handle = document.createElement("div");

            handle.className =
                `resize-handle resize-${dir}`;

            wrapper.appendChild(handle);

            handle.addEventListener("mousedown", e => {

                e.preventDefault();

                const rect =
                    textarea.getBoundingClientRect();

                const start = {
                    mouseX: e.clientX,
                    mouseY: e.clientY,
                    width: rect.width,
                    height: rect.height,
                    left: rect.left,
                    top: rect.top
                };

                textarea.classList.add(
                    "textarea-resizing"
                );

                document.body.style.userSelect = "none";
                document.body.style.cursor =
                    getComputedStyle(handle).cursor;

                function move(ev) {

                    let width = start.width;
                    let height = start.height;
                    let left = start.left;
                    let top = start.top;

                    const dx =
                        ev.clientX - start.mouseX;

                    const dy =
                        ev.clientY - start.mouseY;

                    if (dir.includes("e")) {
                        width = start.width + dx;
                    }

                    if (dir.includes("s")) {
                        height = start.height + dy;
                    }

                    if (dir.includes("w")) {

                        width = start.width - dx;

                        if (width >= MIN_WIDTH) {

                            left =
                                start.left + dx;

                            textarea.style.position =
                                "fixed";

                            textarea.style.left =
                                left + "px";
                        }
                    }

                    if (dir.includes("n")) {

                        height = start.height - dy;

                        if (height >= MIN_HEIGHT) {

                            top =
                                start.top + dy;

                            textarea.style.position =
                                "fixed";

                            textarea.style.top =
                                top + "px";
                        }
                    }

                    textarea.style.width =
                        Math.max(
                            MIN_WIDTH,
                            width
                        ) + "px";

                    textarea.style.height =
                        Math.max(
                            MIN_HEIGHT,
                            height
                        ) + "px";
                }

                function stop() {

                    textarea.classList.remove(
                        "textarea-resizing"
                    );

                    document.body.style.userSelect =
                        "";

                    document.body.style.cursor =
                        "";

                    document.removeEventListener(
                        "mousemove",
                        move
                    );

                    document.removeEventListener(
                        "mouseup",
                        stop
                    );
                }

                document.addEventListener(
                    "mousemove",
                    move
                );

                document.addEventListener(
                    "mouseup",
                    stop
                );
            });
        });
    }

    function init() {
        document
            .querySelectorAll("textarea")
            .forEach(makeResizable);
    }

    init();

    new MutationObserver(init)
        .observe(document.body, {
            childList: true,
            subtree: true
        });

})();
</script>
