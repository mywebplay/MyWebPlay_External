<script>
const placeholders = [
    "Nhập từ khoá tìm kiếm theo địa chỉ IP",
    "Nhập từ khoá tìm kiếm theo host",
    "Nhập từ khoá tìm kiếm theo ngày truy cập (bắt đầu, định dạng: yyyy-MM-dd HH:mm:ss)",
     "Nhập từ khoá tìm kiếm theo ngày truy cập (kết thúc, định dạng: yyyy-MM-dd HH:mm:ss)",
    "Nhập từ khoá tìm kiếm theo thông tin truy cập",
    "Nhập từ khoá tìm kiếm theo trạng thái",
];

for (let i = 0; i < placeholders.length; i++) {
    document.getElementById("txtParam" + i).placeholder = placeholders[i];
    document.getElementById("txtParam" + i).title = placeholders[i];
}

document.getElementById("txtType2").selectedIndex = 5;
document.getElementById("txtType3").selectedIndex = 3;

document.getElementById("txtParam2").readOnly = true;
document.getElementById("txtParam3").readOnly = true;

// ===== GLOBAL =====
let currentInput = null;
let picker = null;

// ===== BIND TEXTBOX =====
function bindDateTimePicker(id) {
    const el = document.getElementById(id);
    if (!el) return;

    el.addEventListener('click', function () {
        openDateTimeModal(el);
    });
}

// ===== OPEN MODAL =====
function openDateTimeModal(input) {
    currentInput = input;

    const modal = document.getElementById('dtModal');
    const chk = document.getElementById('chkUseTime');

    modal.style.display = 'block';

    if (picker) picker.destroy();

   // const hasTime = input.value && input.value.includes(':');
   // chk.checked = hasTime;

    picker = flatpickr("#dtPicker", {
        enableTime: true,
        enableSeconds: true,
        dateFormat: "Y-m-d H:i:S",
        time_24hr: true,
        defaultDate: input.value || null
    });
}

// ===== TOGGLE TIME =====
document.addEventListener('change', function (e) {
    if (e.target && e.target.id === 'chkUseTime') {

        if (!picker) return;

        const selected = picker.selectedDates[0];

        picker.destroy();

        picker = flatpickr("#dtPicker", {
            enableTime: true,
            enableSeconds: true,
            dateFormat: "Y-m-d H:i:S",
            time_24hr: true,
            defaultDate: selected || null
        });
    }
});

// ===== CLICK EVENTS =====
document.addEventListener('click', function (e) {

    // OK
    if (e.target.closest('#btnOk')) {

        if (!picker || !currentInput) return;

        const selected = picker.selectedDates[0];
        if (!selected) return;

        const useTime = document.getElementById('chkUseTime').checked;

        // let value = useTime
           // ? formatDateTime(selected)
           // : formatDate(selected);

        const value = formatDateTime(selected);

        currentInput.value = value;

        closeModal(); // ✅ chỉ đóng khi OK
        return;
    }

    // CANCEL
    if (e.target.closest('#btnCancel')) {
        closeModal(); // ✅ chỉ đóng khi Cancel
        return;
    }

    // CLEAR
    if (e.target.closest('#btnClear')) {

        if (currentInput) {
            currentInput.value = ""; // 🔥 xoá textbox ngoài
        }

        // reset picker luôn cho sạch
        if (picker) {
            picker.clear();
        }

        closeModal();
        return;
    }

    // ❌ KHÔNG có dòng closeModal() ngoài này nữa
});

// ===== CLOSE MODAL =====
function closeModal() {
    const modal = document.getElementById('dtModal');
    if (modal) modal.style.display = 'none';
}

// ===== FORMAT DATE =====
function formatDate(d) {
    const yyyy = d.getFullYear();
    const MM = String(d.getMonth() + 1).padStart(2, '0');
    const dd = String(d.getDate()).padStart(2, '0');

    return `${yyyy}-${MM}-${dd}`;
}

// ===== FORMAT DATETIME =====
function formatDateTime(d) {
    const yyyy = d.getFullYear();
    const MM = String(d.getMonth() + 1).padStart(2, '0');
    const dd = String(d.getDate()).padStart(2, '0');

    const HH = String(d.getHours()).padStart(2, '0');
    const mm = String(d.getMinutes()).padStart(2, '0');
    const ss = String(d.getSeconds()).padStart(2, '0');

    return `${yyyy}-${MM}-${dd} ${HH}:${mm}:${ss}`;
}
