<script>
function SeeTitle(i, nguyennhan, giaiphap)
{
    var row = document.getElementById("tr_" + i);

    if (row == null)
    {
        return;
    }

    var hasNguyenNhan =
        nguyennhan != null &&
        nguyennhan.toString().trim() !== "";

    var hasGiaiPhap =
        giaiphap != null &&
        giaiphap.toString().trim() !== "";

    if (!hasNguyenNhan && !hasGiaiPhap)
    {
        return;
    }

    row.onclick = function ()
    {
        var oldModal = document.getElementById("myModalGlobal");

        if (oldModal)
        {
            oldModal.remove();
        }

        var html = "";

        if (hasNguyenNhan)
        {
            html += `
                <div class="my-modal-section">
                    <h2 class="my-modal-title">
                        NGUYÊN NHÂN PHÁT SINH:
                    </h2>

                    <div class="my-modal-text">
                        ${nguyennhan}
                    </div>
                </div>
            `;
        }

        if (hasGiaiPhap)
        {
            html += `
                <div class="my-modal-section">
                    <h2 class="my-modal-title">
                        GIẢI PHÁP ĐÃ ÁP DỤNG:
                    </h2>

                    <div class="my-modal-text">
                        ${giaiphap}
                    </div>
                </div>
            `;
        }

        var modal = document.createElement("div");

        modal.id = "myModalGlobal";

        modal.innerHTML = `
            <div class="my-modal-overlay">
                <div class="my-modal-box">

                    <button
                        type="button"
                        class="my-modal-close">
                        ✕
                    </button>

                    <div class="my-modal-content">
                        ${html}
                    </div>

                </div>
            </div>
        `;

        document.body.appendChild(modal);

        modal
            .querySelector(".my-modal-close")
            .onclick = function ()
            {
                modal.remove();
            };
    };
}

    SeeTitle(25, "Setting sai thành phần phân tách (hãy chú ý vào đoạn: ---).", null);
    SeeTitle(40, "Logic hiện tại chỉ gửi result mail cho operator khi quá trình send e-mail xảy ra exception/error ngoài ý muốn.", null);
    SeeTitle(57, "Kiểm nha nhầm/sót: một số câu hỏi trắc nghiệm và đáp án vẫn nằm trong phạm vi setting (TH1 hoặc TH2).", null);
    SeeTitle(61, "Setting sai: trang web đang yêu cầu nhập key là web sub nhưng lại set nhầm cookie ở web normal.", null);
    SeeTitle(63, "Các url externalX nếu nằm trong list URL cấm vào chỉ khi IP đang truy cập không phải là admin hoặc setting NOT_IS_ADMINUSING.", null);
    SeeTitle(65, "Do quên bổ sung url mới vào file ListPageSetting.txt.", null);
</script>
