function sendmail()
{
    localStorage.setItem(
        "data_subject",
        "{" + window.location.hostname + "} - Backup dữ liệu ? - từ DX về cho admin"
    );

    var data = document.getElementById("tableContainer").outerHTML;

    data = data.replaceAll("</th>", "@@TH@@");
    data = data.replaceAll("</td>", "@@TD@@");

    data = data.replaceAll("<th", "<th style=\"border:1px solid black;\"");
    data = data.replaceAll("<td", "<td style=\"border:1px solid black;\"");

    data = data.replaceAll("@@TH@@", "</th>");
    data = data.replaceAll("@@TD@@", "</td>");

    data = data.replaceAll(
        "<th style=\"border:1px solid black;\"ead>",
        "<thead>"
    );

    localStorage.setItem("data_message", data);
    localStorage.setItem("data_isHTML", "true");

    window.open("/Cover/CustomerPagePlay?code=sendmail", "_blank");
}



function downloadCSV()
{
    var table = document.querySelector("#tableContainer table");

    if (!table)
    {
        alert("Không tìm thấy table trong #tableContainer");
        return;
    }

    var rows = table.querySelectorAll("tr");
    var csv = [];

    rows.forEach(function(row)
    {
        var cols = row.querySelectorAll("th, td");
        var rowData = [];

        cols.forEach(function(col)
        {
            var text = "";

            var input = col.querySelector("input, textarea, select");

            if (input)
            {
                if (input.tagName === "SELECT")
                {
                    var opt = input.options[input.selectedIndex];
                    text = opt ? opt.text : "";
                }
                else
                {
                    text = input.value || "";
                }
            }
            else
            {
                text = col.innerText || col.textContent || "";
            }

            text = text.replace(/"/g, '""');

            rowData.push('"' + text + '"');
        });

        csv.push(rowData.join(","));
    });

    var BOM = "\uFEFF";

    var blob = new Blob(
        [BOM + csv.join("\n")],
        { type: "text/csv;charset=utf-8;" }
    );

    var url = URL.createObjectURL(blob);

    var a = document.createElement("a");
    a.href = url;
    a.download = "data.csv";
    a.click();
}

document.addEventListener('DOMContentLoaded', function () {

    const pageSize = 50;
    const totalRecord = 135;
    const totalPage = Math.max(1, Math.ceil(totalRecord / pageSize));

    const params = new URLSearchParams(window.location.search);
    let page = parseInt(params.get('page'), 10);

    if (isNaN(page) || page < 1) page = 1;
    if (page > totalPage) page = totalPage;

    document.querySelectorAll('.pagingBar').forEach(bar => {

        const btnPrev = bar.querySelector('.btnPrev');
        const btnNext = bar.querySelector('.btnNext');
        const pageNumber = bar.querySelector('.pageNumber');

        if (pageNumber) {
            pageNumber.textContent = page;
        }

        // ===== PREV =====
        if (btnPrev) {
            btnPrev.disabled = (page <= 1);
            btnPrev.onclick = function () {
                if (btnPrev.disabled) return;
                const u = new URL(window.location.href);
                u.searchParams.set('page', page - 1);
                window.location.href = u.toString();
            };
        }

        // ===== NEXT =====
        if (btnNext) {
            btnNext.disabled = (page >= totalPage);
            btnNext.onclick = function () {
                if (btnNext.disabled) return;
                const u = new URL(window.location.href);
                u.searchParams.set('page', page + 1);
                window.location.href = u.toString();
            };
        }
    });
});

function TimKiem() {
    var parax = "";
    var hasQuery = window.location.search.length > 0; // 🔑 quyết định ? hay &

    // xoá param= và page= khỏi URL hiện tại (GIỮ query khác)
    var search = window.location.search
        .replace(/([?&])(param|page|type|orderBy)=[^&]*/gi, "")
        .replace(/^&/, "?")
        .replace(/\?&/, "?")
        .replace(/\?$/, "");

    hasQuery = search.length > 0;

    for (var i = 0; i < 6; i++) {
        var value = document.getElementById("txtParam" + i).value;
        var value2 = document.getElementById("txtType" + i).value;
        parax += (hasQuery ? "&" : "?") + "param=" + encodeURIComponent(value) + "&type=" + encodeURIComponent(value2);
        hasQuery = true; // 🔑 sau lần đầu thì chắc chắn là &
    }

    // redirect
    var value3 = document.getElementById("option_sort").value;
    window.location.href = window.location.pathname + search + parax  + "&orderBy=" + encodeURIComponent(value3);
}

function ClearSearch() {
    const params = new URLSearchParams(window.location.search);
    const code = params.get("code");
    window.location.href = window.location.pathname + "?code=" + code;
}
