/* ===== SENDMAIL GIỮ NGUYÊN ===== */
async function sendmail(){
showProcessingModal();

var formData = new FormData();
formData.append("txtEmail", mail.value);
formData.append("txtMessage", message.value);
formData.append("txtSubject", subject.value);
formData.append("isBodyHTML", "" + isHTML.checked);
formData.append("indexAnotherLoginMail", "" + indexAnotherLoginMail.value);

for (var i = 0; i < files.files.length; i++) {
  formData.append('files', files.files[i]);
}

const res = await fetch("/Home/SendMailSave1?isUploadFileMegaIO=" + (sendFile.checked?"true":"false"), {
method:"POST",
body:formData
});
const data = await res.json();

if(data.result==true){
closeProcessingModal();
alert("Tiến trình gửi e-mail hiện đang được xử lý!");
}
return data;
}

function parseCSV(text){
  const rows=text.split("\n").filter(r=>r.trim());
  const headers=rows[0].split(",");
  return rows.slice(1).map(r=>{
    const cols=r.split(",");
    let obj={};
    headers.forEach((h,i)=>obj[h.trim()]=cols[i]?.trim());
    return obj;
  });
}

function replaceTemplate(str,data){
  return str.replace(/@@\s*(.*?)\s*@@/g,(_,k)=>data[k]||"");
}

/* ===== BULK ===== */
async function sendBulkMail(){

if(!confirm("Bạn có chắc muốn gửi hàng loạt?")) return;

const file = csvFile.files[0];
if(!file || !file.name.toLowerCase().endsWith(".csv")){
  alert("Vui lòng chọn file csv!");
  return;
}

const text = await file.text();
const list = parseCSV(text);

let success=0, error=0;

bulkOverlay.style.display="flex";

/* CHẶN USER */
document.body.style.overflow="hidden";
document.body.style.pointerEvents="none";
bulkOverlay.style.pointerEvents="auto";

for(let i=0;i<list.length;i++){

const item=list[i];
const email=item[Object.keys(item)[0]];

currentEmail.innerText="Đang gửi: "+email;

mail.value=email;
subject.value=replaceTemplate(subject_bulk.value,item);
message.value=replaceTemplate(message_bulk.value,item);

isHTML.checked=isHTML_bulk.checked;
sendFile.checked=sendFile_bulk.checked;
indexAnotherLoginMail.value=indexAnotherLoginMail_bulk.value;

/* copy file */
if(files_bulk.files.length>0){
  const dt=new DataTransfer();
  for(let f of files_bulk.files) dt.items.add(f);
  files.files=dt.files;
}

const res=await sendmail();
res.result?success++:error++;

let percent=Math.round((i+1)/list.length*100);
bulkBar.style.width=percent+"%";
bulkPercent.innerText=percent+"%";
successCount.innerText=success;
errorCount.innerText=error;

await new Promise(r=>setTimeout(r,5000));
}

bulkOverlay.style.display="none";
document.body.style.overflow="";
document.body.style.pointerEvents="";

alert("Đã gửi email hàng loạt!\nSuccess: "+success+"\nError: "+error);
}

/* ===== PROCESS TAB 1 ===== */
let processingModalVisible=false;
function showProcessingModal(){
if(processingModalVisible)return;
processingOverlay.classList.add("active");
document.body.style.overflow="hidden";
document.body.style.pointerEvents="none";
processingOverlay.style.pointerEvents="auto";
processingModalVisible=true;
}
function closeProcessingModal(){
if(!processingModalVisible)return;
processingOverlay.classList.remove("active");
document.body.style.overflow="";
document.body.style.pointerEvents="";
processingModalVisible=false;
}
