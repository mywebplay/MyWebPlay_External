<script>
function switchTab(tab) {
  document.getElementById("tab1").classList.toggle("hidden", tab !== 1);
  document.getElementById("tab2").classList.toggle("hidden", tab !== 2);
  document.getElementById("tabBtn1").classList.toggle("active", tab === 1);
  document.getElementById("tabBtn2").classList.toggle("active", tab === 2);
}

function parseCSV(text) {
  const rows = text.split("\n").map(r => r.trim()).filter(r => r);
  const headers = rows[0].split(",");
  const data = [];

  for (let i = 1; i < rows.length; i++) {
    const cols = rows[i].split(",");
    let obj = {};
    headers.forEach((h, idx) => obj[h.trim()] = cols[idx]?.trim());
    data.push(obj);
  }
  return data;
}

function replaceTemplate(str, data) {
  return str.replace(/@@\s*(.*?)\s*@@/g, (_, key) => data[key] || "");
}

async function sendBulkMail() {
  const file = document.getElementById("csvFile").files[0];
  if (!file) return alert("Chọn file CSV");

  const text = await file.text();
  const list = parseCSV(text);

  let success = 0, error = 0;
  const total = list.length;

  document.getElementById("processingOverlay").classList.add("active");

  for (let i = 0; i < total; i++) {
    const item = list[i];

    const subject = replaceTemplate(document.getElementById("subject_bulk").value, item);
    const message = replaceTemplate(document.getElementById("message_bulk").value, item);

    try {
      let formData = new FormData();
      formData.append("txtEmail", item[Object.keys(item)[0]]);
      formData.append("txtSubject", subject);
      formData.append("txtMessage", message);
      formData.append("isBodyHTML", "false");

      const res = await fetch("/Home/SendMailSave1", {
        method: "POST",
        body: formData
      });

      const data = await res.json();
      if (data.result) success++; else error++;
    } catch {
      error++;
    }

    document.getElementById("successCount").innerText = success;
    document.getElementById("errorCount").innerText = error;
    document.getElementById("progressBar").style.width = ((i + 1) / total * 100) + "%";

    var percent = Math.round((i + 1) / total * 100);

    var progressBar = document.getElementById("progressBar");
    var progressText = document.getElementById("progressText");
    
    progressBar.style.width = percent + "%";
    progressText.innerText = percent + "%";

    await new Promise(r => setTimeout(r, 5000));
  }

  document.getElementById("processingOverlay").classList.remove("active");

  alert(`Đã gửi email hàng loạt!\nSuccess: ${success}\nError: ${error}`);
}
</script>
