<script>
function switchTab(tab){
  document.getElementById("tab1").style.display = (tab===1?"block":"none");
  document.getElementById("tab2").style.display = (tab===2?"block":"none");
}

function showModalView(){
  document.getElementById("modal_data").innerHTML =
      document.getElementById("message").value;
  document.getElementById("modal_view").style.display = "block";
}
function previewBulk(){
  document.getElementById("modal_data").innerHTML =
      document.getElementById("message_bulk").value;
  document.getElementById("modal_view").style.display = "block";
}
function closeModalView(){
  document.getElementById("modal_view").style.display = "none";
}

async function sendmail(){
  var formData = new FormData();
  formData.append("txtEmail", document.getElementById("mail").value);
  formData.append("txtSubject", document.getElementById("subject").value);
  formData.append("txtMessage", document.getElementById("message").value);

  const res = await fetch("/Home/SendMailSave1", {
    method: "POST",
    body: formData
  });

  return res.json();
}

function parseCSV(text){
  const rows = text.split("\n").filter(r=>r.trim());
  const headers = rows[0].split(",");
  return rows.slice(1).map(r=>{
    const cols = r.split(",");
    let obj = {};
    headers.forEach((h,i)=>obj[h.trim()] = cols[i]?.trim());
    return obj;
  });
}

function replaceTemplate(str,data){
  return str.replace(/@@\\s*(.*?)\\s*@@/g,(_,k)=>data[k]||"");
}

async function sendBulkMail(){
  const file = document.getElementById("csvFile").files[0];
  if(!file) return alert("Chọn CSV");

  const text = await file.text();
  const list = parseCSV(text);

  let success=0, error=0;
  const total=list.length;

  document.getElementById("processingOverlay").style.display="block";

  for(let i=0;i<total;i++){
    const item=list[i];

    document.getElementById("mail").value = item[Object.keys(item)[0]];
    document.getElementById("subject").value =
      replaceTemplate(document.getElementById("subject_bulk").value,item);
    document.getElementById("message").value =
      replaceTemplate(document.getElementById("message_bulk").value,item);

    document.getElementById("isHTML").checked =
      document.getElementById("isHTML_bulk").checked;

    document.getElementById("sendFile").checked =
      document.getElementById("sendFile_bulk").checked;

    document.getElementById("indexAnotherLoginMail").value =
      document.getElementById("indexAnotherLoginMail_bulk").value;

    const bulkFiles = document.getElementById("files_bulk").files;
    if(bulkFiles.length>0){
      const dt = new DataTransfer();
      for(let f of bulkFiles) dt.items.add(f);
      document.getElementById("files").files = dt.files;
    }

    try{
      const res = await sendmail();
      if(res && res.result) success++; else error++;
    }catch{
      error++;
    }

    document.getElementById("successCount").innerText = success;
    document.getElementById("errorCount").innerText = error;

    var percent = Math.round((i+1)/total*100);
    document.getElementById("progressBar").style.width = percent + "%";
    document.getElementById("progressText").innerText = percent + "%";

    await new Promise(r=>setTimeout(r,5000));
  }

  document.getElementById("processingOverlay").style.display="none";

  alert("Đã gửi email hàng loạt!\nSuccess: "+success+"\nError: "+error);
}
</script>
