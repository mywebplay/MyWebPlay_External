    <script>
        document.getElementById("fileForm").addEventListener("submit", async function(e) {
         e.preventDefault();
        var confirmX = confirm("Bạn có chắc chắn không? Kiểm tra kỹ mọi hành động trước khi OK nhé...");
        if (confirmX == false) return;
            
            const form = e.target;
            const formData = new FormData(form);

            const action = formData.get("Action");
            const res = await fetch("/Cover/ExectFilePower", {
                method: "POST",
                body: formData
            });

            const data = await res.json();

            // Ẩn vùng kết quả trước
            document.getElementById("resultArea").style.display = "none";
            document.getElementById("downloadLink").style.display = "none";

            if (action === "READ" && data.noidung !== undefined) {
                document.getElementById("resultText").value = data.noidung;
                document.getElementById("resultArea").style.display = "block";
            }
            else if ((action === "LIST-FILE" || action === "LIST-FOLDER") && data.list !== undefined) {
                document.getElementById("resultText").value = data.list;
                var count = document.getElementById("resultText").value.split('\n').length;
                document.getElementById("count").innerText = "("+count+")";
                document.getElementById("resultArea").style.display = "block";
            }
            else if (action === "DOWNLOAD" && data.link !== undefined) {
                document.getElementById("downloadA").href = data.link;
                document.getElementById("downloadLink").style.display = "block";
            }
            else if (data.result === true) {
                alert("Successful!");
            }
            else {
                alert("Failed!");
            }
        });
    </script>
